req = {"id": [data_id],
       "timestamp": [timestamp],
       "vec": [noisy_data.tolist()]}

resp = requests.post(f'{DEEPDENOISER_API_URL}/predict', json=req)
print(resp)

denoised_data = np.array(resp.json()["vec"])
